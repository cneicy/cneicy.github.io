#include<iostream>
#include<algorithm>
using namespace std;
int g[110][110];
int main(){
    int n,a,b;cin>>n>>a>>b;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(i==j) g[i][j]=0;
            else g[i][j]=1e8;
        }
    }
    for(int i=1;i<=n;i++){
        int x;cin>>x;
        for(int j=1;j<=x;j++){
            int y;cin>>y;
            if(j==1) g[i][y]=0;
            else g[i][y]=1;
        }
    }
    for(int k=1;k<=n;k++){
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++){
                if(k!=i&&i!=j&&j!=k)
                    g[i][j]=min(g[i][j],g[i][k]+g[k][j]);
            }
        }
    }
    cout<<g[a][b];
}
