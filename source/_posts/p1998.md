#include<iostream>
#include<string>
using namespace std;
int a[1000],c[1000],len;
int main()
{
    int n;cin>>n;
    a[1]=1,len=1;
    for(int i=1;i<=n;i++)
    {    /*
        1(a[1])2(a[2])0(a[3])
        */
        for(int j=1;j<=len;j++)
        {
            a[j]*=i;//阶乘 
        }
        for(int i=1;i<=len;i++)
        {
            a[i+1]+=a[i]/10;a[i]%=10;//进位 
            if(a[len+1]>0) len++;//长度判断 
        }
        for(int i=1;i<=len;i++)
        {
            c[i]+=a[i];//保留相加 
            c[i+1]+=c[i]/10;//进位 
            c[i]=c[i]%10;//进位 
            if(c[len+1]) len++;//长度判断 
        }
    }
    for(int i=len;i>=1;i--) cout<<c[i];//倒序输出 
    return 0;
}
